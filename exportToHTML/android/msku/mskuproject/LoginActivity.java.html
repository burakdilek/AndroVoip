121      private boolean mayRequestContacts() { 
122          if (Build.VERSION.SDK_INT < Build.VERSION_CODES.M) { 
123              return true; 
124          } 
125          if (checkSelfPermission(READ_CONTACTS) == PackageManager.PERMISSION_GRANTED) { 
126              return true; 
127          } 
128          if (shouldShowRequestPermissionRationale(READ_CONTACTS)) { 
129              Snackbar.make(mEmailView, R.string.permission_rationale, Snackbar.LENGTH_INDEFINITE) 
130                      .setAction(android.R.string.ok, new View.OnClickListener() { 
131                          @Override 
132                          @TargetApi(Build.VERSION_CODES.M) 
133                          public void onClick(View v) { 
134                              requestPermissions(new String[]{READ_CONTACTS}, REQUEST_READ_CONTACTS); 
135                          } 
136                      }); 
137          } else { 
138              requestPermissions(new String[]{READ_CONTACTS}, REQUEST_READ_CONTACTS); 
139          } 
140          return false; 
141      }

157      /** 
158       * Attempts to sign in or register the account specified by the login form. 
159       * If there are form errors (invalid email, missing fields, etc.), the 
160       * errors are presented and no actual login attempt is made. 
161       */ 
162      private void attemptLogin() { 
163          if (mAuthTask != null) { 
164              return; 
165          } 
166   
167          // Reset errors. 
168          mEmailView.setError(null); 
169          mPasswordView.setError(null); 
170   
171          // Store values at the time of the login attempt. 
172          String email = mEmailView.getText().toString(); 
173          String password = mPasswordView.getText().toString(); 
174          String domain = mDomainView.getText().toString(); 
175          String port = mPortView.getText().toString(); 
176          String proxy = mProxyView.getText().toString(); 
177   
178          boolean cancel = false; 
179          View focusView = null; 
180   
181          // Check for a valid password, if the user entered one. 
182          if (!TextUtils.isEmpty(password) && !isPasswordValid(password)) { 
183              mPasswordView.setError(getString(R.string.error_invalid_password)); 
184              focusView = mPasswordView; 
185              cancel = true; 
186          } 
187   
188          // Check for a valid email address. 
189          if (TextUtils.isEmpty(email)) { 
190              mEmailView.setError(getString(R.string.error_field_required)); 
191              focusView = mEmailView; 
192              cancel = true; 
193          } 
194          if (cancel) { 
195              // There was an error; don't attempt login and focus the first 
196              // form field with an error. 
197              focusView.requestFocus(); 
198          } else { 
199              // Show a progress spinner, and kick off a background task to 
200              // perform the user login attempt. 
201   
202              showProgress(true); 
203              mAuthTask = new UserLoginTask(email, password, domain, port, proxy); 
204              mAuthTask.execute((Void) null); 
205          } 
206      }

218      @TargetApi(Build.VERSION_CODES.HONEYCOMB_MR2) 
219      private void showProgress(final boolean show) { 
220          // On Honeycomb MR2 we have the ViewPropertyAnimator APIs, which allow 
221          // for very easy animations. If available, use these APIs to fade-in 
222          // the progress spinner. 
223          if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.HONEYCOMB_MR2) { 
224              int shortAnimTime = getResources().getInteger(android.R.integer.config_shortAnimTime); 
225   
226              mLoginFormView.setVisibility(show ? View.GONE : View.VISIBLE); 
227              mLoginFormView.animate().setDuration(shortAnimTime).alpha( 
228                      show ? 0 : 1).setListener(new AnimatorListenerAdapter() { 
229                  @Override 
230                  public void onAnimationEnd(Animator animation) { 
231                      mLoginFormView.setVisibility(show ? View.GONE : View.VISIBLE); 
232                  } 
233              }); 
234   
235              mProgressView.setVisibility(show ? View.VISIBLE : View.GONE); 
236              mProgressView.animate().setDuration(shortAnimTime).alpha( 
237                      show ? 1 : 0).setListener(new AnimatorListenerAdapter() { 
238                  @Override 
239                  public void onAnimationEnd(Animator animation) { 
240                      mProgressView.setVisibility(show ? View.VISIBLE : View.GONE); 
241                  } 
242              }); 
243          } else { 
244              // The ViewPropertyAnimator APIs are not available, so simply show 
245              // and hide the relevant UI components. 
246              mProgressView.setVisibility(show ? View.VISIBLE : View.GONE); 
247              mLoginFormView.setVisibility(show ? View.GONE : View.VISIBLE); 
248          } 
249      }

252      @Override 
253      public Loader<Cursor> onCreateLoader(int i, Bundle bundle) { 
254          return new CursorLoader(this, 
255                  // Retrieve data rows for the device user's 'profile' contact. 
256                  Uri.withAppendedPath(ContactsContract.Profile.CONTENT_URI, 
257                          ContactsContract.Contacts.Data.CONTENT_DIRECTORY), ProfileQuery.PROJECTION, 
258   
259                  // Select only email addresses. 
260                  ContactsContract.Contacts.Data.MIMETYPE + 
261                          " = ?", new String[]{ContactsContract.CommonDataKinds.Email 
262                  .CONTENT_ITEM_TYPE}, 
263   
264                  // Show primary email addresses first. Note that there won't be 
265                  // a primary email address if the user hasn't specified one. 
266                  ContactsContract.Contacts.Data.IS_PRIMARY + " DESC"); 
267      }

269      @Override 
270      public void onLoadFinished(Loader<Cursor> cursorLoader, Cursor cursor) { 
271          List<String> emails = new ArrayList<>(); 
272          cursor.moveToFirst(); 
273          while (!cursor.isAfterLast()) { 
274              emails.add(cursor.getString(ProfileQuery.ADDRESS)); 
275              cursor.moveToNext(); 
276          } 
277   
278          addEmailsToAutoComplete(emails); 
279      }

297       * Use an AsyncTask to fetch the user's email addresses on a background thread, and update 
298       * the email text field with results on the main UI thread. 
299       */ 
300      class SetupEmailAutoCompleteTask extends AsyncTask<Void, Void, List<String>> { 
301   
302          @Override 
303          protected List<String> doInBackground(Void... voids) { 
304              ArrayList<String> emailAddressCollection = new ArrayList<>(); 
305   
306              // Get all emails from the user's contacts and copy them to a list. 
307              ContentResolver cr = getContentResolver(); 
308              Cursor emailCur = cr.query(ContactsContract.CommonDataKinds.Email.CONTENT_URI, null, 
309                      null, null, null); 
310              while (emailCur.moveToNext()) { 
311                  String email = emailCur.getString(emailCur.getColumnIndex(ContactsContract 
312                          .CommonDataKinds.Email.DATA)); 
313                  emailAddressCollection.add(email); 
314              } 
315              emailCur.close(); 
316   
317              return emailAddressCollection; 
318          } 
319   
320          @Override 
321          protected void onPostExecute(List<String> emailAddressCollection) { 
322              addEmailsToAutoComplete(emailAddressCollection); 
323          } 
324      }

329                  new ArrayAdapter<>(LoginActivity.this, 
330                          android.R.layout.simple_dropdown_item_1line, emailAddressCollection); 
331   
332          mEmailView.setAdapter(adapter);
